    {
        switch (activationMode)
        {
            case TrapActivationMode.AllAtOnce:
                ActivateAllTraps();

            case TrapActivationMode.Sequential:
                StartCoroutine(ActivateSequentially());
                break;
                
            case TrapActivationMode.OnPlayerEnter:
